       ```python
       def _get_flat_params(model: nn.Module) -> torch.Tensor:
           """Flatten all trainable parameters into a single 1D tensor."""
           return torch.cat([p.reshape(-1) for p in model.parameters() if p.requires_grad])

       def _set_flat_params(model: nn.Module, flat_params: torch.Tensor) -> None:
           """Set model parameters from a flat 1D tensor."""
           offset = 0
           for p in model.parameters():
               if p.requires_grad:
                   numel = p.numel()
                   p.data.copy_(flat_params[offset:offset+numel].reshape(p.shape))
                   offset += numel

       def _unflatten_like(flat_vector: torch.Tensor, params: list[torch.nn.Parameter]) -> list[torch.Tensor]:
           """Reshape flat vector into list of tensors matching param shapes."""
           views = []
           offset = 0
           for p in params:
               numel = p.numel()
               views.append(flat_vector[offset:offset+numel].reshape(p.shape))
               offset += numel
           return views

       def compute_full_hessian(
           model: nn.Module,
           loss_fn: Callable,
           data_x: torch.Tensor,
           data_y: torch.Tensor,
           device: str | torch.device = "cpu",
       ) -> dict[str, Any]:
       ```
       - Collect all trainable params, flatten
       - Define scalar loss function of flat params: unflatten -> set params -> forward -> loss
       - Call torch.autograd.functional.hessian(loss_func, flat_params)
       - Compute eigenvalues via torch.linalg.eigvalsh(H)
       - Return dict with: eigenvalues (numpy array), n_negative, n_positive, n_zero (within 1e-10), trace, spectral_norm, negative_ratio, method='full'

       ```python
       def hessian_vector_product(
           grad_flat: torch.Tensor,
           params: list[nn.Parameter],
           v_list: list[torch.Tensor],
       ) -> torch.Tensor:
       ```
       - Compute Hv via torch.autograd.grad(grad_flat, params, grad_outputs=v_list, retain_graph=True)
       - Return concatenated flat Hv

       ```python
       def stochastic_lanczos(
           model: nn.Module,
           loss_fn: Callable,
           data_x: torch.Tensor,
           data_y: torch.Tensor,
           n_iterations: int = 200,
           n_samples: int = 5,
           device: str | torch.device = "cpu",
       ) -> dict[str, Any]:
       ```
       - Full reorthogonalization Lanczos iteration (Ghorbani et al. 2019)
       - For each sample: random unit starting vector v0, run Lanczos for n_iterations
       - At each step k: compute Hv_k via HVP, compute alpha_k = v_k @ Hv_k, update w = Hv_k - alpha_k*v_k - beta_{k-1}*v_{k-1}
       - Full reorthogonalization: for each previous v_j, subtract (w @ v_j) * v_j from w
       - Build tridiagonal matrix T, compute eigenvalues (Ritz values)
       - IMPORTANT: compute gradient with create_graph=True once, reuse for all HVPs
       - Return dict with: ritz_values (numpy array of all Ritz values), n_negative_approx, negative_ratio_approx, trace_approx, method='lanczos', n_iterations, n_samples

       ```python
       def compute_hessian_spectrum(

           data_x: torch.Tensor,
           data_y: torch.Tensor,
           device: str | torch.device = "cpu",
           method: str = "auto",
           max_full_params: int = 2000,
       ) -> dict[str, Any]:
       ```
       - If method='auto': use 'full' if n_params <= max_full_params else 'lanczos'
       - Dispatches to compute_full_hessian or stochastic_lanczos

       ```python
       def _filter_normalize(direction: dict[str, torch.Tensor], model: nn.Module) -> None:
           """Apply Li et al. 2018 filter normalization in-place.

           For each parameter with dim >= 2, scale each filter (row) of the direction
           to match the corresponding filter's norm in the model parameters.
           For 1D params (bias), normalize to match parameter norm.
           """
           for name, param in model.named_parameters():
               d = direction[name]
               if param.dim() >= 2:
                   # Per-filter normalization
                   for j in range(d.shape[0]):
                       d[j] *= param[j].norm() / (d[j].norm() + 1e-10)
               else:
                   # Scalar/bias: match overall norm
                   d *= param.norm() / (d.norm() + 1e-10)

       def _evaluate_loss(
           model: nn.Module,
           loss_fn: Callable,
           data_x: torch.Tensor,
           data_y: torch.Tensor,
       ) -> float:
           """Evaluate model loss on data batch (no grad)."""
           model.eval()
           with torch.no_grad():
               output = model(data_x)
               return loss_fn(output, data_y).item()

       def measure_curvature(
           model: nn.Module,
           loss_fn: Callable,
           data_x: torch.Tensor,
           data_y: torch.Tensor,
           n_directions: int = 50,
           n_steps: int = 51,
           step_range: float = 1.0,
           seed: int = 42,
       ) -> dict[str, Any]:
       ```
       - Save converged weights as theta_star = {name: p.clone() for name, p in model.named_parameters()}
       - For each of n_directions random directions:
         a. Generate random direction d = {name: torch.randn_like(p)} seeded by seed+i
         b. Apply filter normalization via _filter_normalize(d, model)
         c. Sample loss at n_steps points: alpha in linspace(-step_range, step_range, n_steps)
         d. At each alpha: set params to theta_star[name] + alpha * d[name], evaluate loss
         e. Fit quadratic via np.polyfit(alphas, losses, 2); curvature = 2 * coeffs[0]
       - Restore original weights from theta_star
       - Return dict with: mean_curvature, median_curvature, std_curvature, curvatures (list), n_directions
